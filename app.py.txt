# --- MOTORE DI CALCOLO ---

def calcoli_avanzati_piva(compenso_base, apply_rivalsa, apply_bollo, aliquota_imp):
    """Calcola tutto partendo dal compenso richiesto."""
    params = PARAMETRI_SPORT["P.IVA Forfettario"]
    
    # 1. Determinazione Fatturato Lordo
    rivalsa_val = compenso_base * 0.04 if apply_rivalsa else 0.0
    fatturato_lordo = compenso_base + rivalsa_val
    bollo_val = 2.0 if apply_bollo and fatturato_lordo > 77.47 else 0.0
    
    # 2. Calcolo Reddito Forfettario
    reddito_forfettario = fatturato_lordo * params["coeff_redditivita"]
    
    # 3. Calcolo INPS (Soglia 5000 + Abbattimento 50%)
    imponibile_inps = reddito_forfettario - params["soglia_inps"]
    if imponibile_inps < 0: imponibile_inps = 0
    imponibile_inps_ridotto = imponibile_inps * params["abbattimento_inps"]
    inps_dovuta = imponibile_inps_ridotto * params["aliquota_inps"]
    
    # 4. Calcolo Tasse (Deduzione Contributi E Franchigia 15.000‚Ç¨)
    # Riforma Sport: La franchigia 15k si applica anche ai forfettari sull'imponibile
    imponibile_fiscale = reddito_forfettario - inps_dovuta - params["soglia_no_tax"]
    if imponibile_fiscale < 0: imponibile_fiscale = 0
    
    tasse = imponibile_fiscale * aliquota_imp
    
    # 5. Netto
    # Netto = Fatturato + Bollo(incassato) - INPS - Tasse - Bollo(versato)
    # Semplificando: Netto = Fatturato - INPS - Tasse
    netto = fatturato_lordo - inps_dovuta - tasse
    
    return {
        "compenso": compenso_base,
        "rivalsa": rivalsa_val,
        "fatturato": fatturato_lordo,
        "bollo": bollo_val,
        "reddito_forf": reddito_forfettario,
        "base_inps": imponibile_inps_ridotto,
        "inps": inps_dovuta,
        "base_fiscale": imponibile_fiscale,
        "tasse": tasse,
        "netto": netto
    }

def calcolo_inverso_piva(netto_target, apply_rivalsa, apply_bollo, aliquota_imp):
    """Trova il Compenso Base partendo dal Netto desiderato (Binary Search)."""
    low = 0.0
    high = netto_target * 2.0 # Stima iniziale ampia
    tolerance = 0.01
    
    for _ in range(100): # Max 100 iterazioni
        mid = (low + high) / 2
        res = calcoli_avanzati_piva(mid, apply_rivalsa, apply_bollo, aliquota_imp)
        diff = res['netto'] - netto_target
        
        if abs(diff) < tolerance:
            return res
        elif diff > 0:
            high = mid
        else:
            low = mid
    return calcoli_avanzati_piva(high, apply_rivalsa, apply_bollo, aliquota_imp)

def calcola_cococo(lordo):
    params = PARAMETRI_SPORT["Collaboratore (Co.co.co)"]
    eccedenza_inps = max(0, lordo - params["soglia_no_inps"])
    base_inps = eccedenza_inps * params["abbattimento_inps"]
    inps_lav = (base_inps * params["aliquota_inps_tot"]) * params["quota_lavoratore"]
    
    imponibile_fiscale = max(0, (lordo - inps_lav) - params["soglia_no_tax"])
    irpef = imponibile_fiscale * 0.23
    
    return {
        "lordo": lordo, "inps": inps_lav, "irpef": irpef, 
        "netto": lordo - inps_lav - irpef
    }

def create_pdf(dati):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", size=12)
    
    # Header Studio
    pdf.set_font("Arial", 'B', 16)
    pdf.set_text_color(0, 42, 82) # Blu
    pdf.cell(200, 10, txt="STUDIO GAETANI - Sport Tax Advisor", ln=1, align='C')
    pdf.set_font("Arial", 'I', 10)
    pdf.set_text_color(100, 100, 100)
    pdf.cell(200, 10, txt="Documento Pro-Forma / Fattura di Cortesia", ln=1, align='C')
    pdf.line(10, 30, 200, 30)
    pdf.ln(10)
    
    # Dati Fornitore/Cliente
    pdf.set_text_color(0, 0, 0)
    pdf.set_font("Arial", 'B', 11)
    pdf.cell(95, 10, txt="FORNITORE:", border=0)
    pdf.cell(95, 10, txt="CLIENTE:", border=0, ln=1)
    
    pdf.set_font("Arial", '', 11)
    pdf.multi_cell(95, 5, txt=dati['mittente'])
    # Spostamento cursore per il cliente
    xy_fornitore = pdf.get_y()
    pdf.set_xy(105, 40) # Torna su per la colonna destra
    pdf.multi_cell(95, 5, txt=dati['destinatario'])
    
    pdf.set_xy(10, max(xy_fornitore, pdf.get_y()) + 10)
    
    # Dati Documento
    pdf.set_fill_color(240, 240, 240)
    pdf.cell(200, 10, txt=f"Documento n. {dati['numero']} del {dati['data']}", ln=1, fill=True)
    pdf.ln(5)
    
    # Tabella Importi
    pdf.set_font("Arial", 'B', 11)
    pdf.cell(130, 10, txt="Descrizione", border=1)
    pdf.cell(60, 10, txt="Importo", border=1, align='R', ln=1)
    
    pdf.set_font("Arial", '', 11)
    pdf.cell(130, 10, txt=dati['descrizione'], border=1)
    pdf.cell(60, 10, txt=f"EUR {dati['compenso']:,.2f}", border=1, align='R', ln=1)
    
    if dati['rivalsa'] > 0:
        pdf.cell(130, 10, txt="Rivalsa INPS 4% (L. 662/96)", border=1)
        pdf.cell(60, 10, txt=f"EUR {dati['rivalsa']:,.2f}", border=1, align='R', ln=1)
        
    # Totali
    pdf.ln(5)
    pdf.set_font("Arial", 'B', 12)
    pdf.cell(130, 10, txt="TOTALE LORDO", border=0, align='R')
    pdf.set_text_color(0, 42, 82)
    pdf.cell(60, 10, txt=f"EUR {dati['totale_lordo']:,.2f}", border=1, align='R', ln=1)
    
    if dati['bollo'] > 0:
        pdf.set_text_color(0, 0, 0)
        pdf.set_font("Arial", '', 10)
        pdf.cell(130, 10, txt="Imposta di Bollo (art. 15 DPR 633/72)", border=0, align='R')
        pdf.cell(60, 10, txt=f"EUR {dati['bollo']:,.2f}", border=0, align='R', ln=1)
        pdf.set_font("Arial", 'B', 14)
        pdf.cell(130, 10, txt="TOTALE A PAGARE", border=0, align='R')
        pdf.cell(60, 10, txt=f"EUR {dati['totale_pagare']:,.2f}", border=0, align='R', ln=1)
    
    # Footer Legale
    pdf.ln(20)
    pdf.set_font("Arial", '', 8)
    pdf.set_text_color(100, 100, 100)
    pdf.multi_cell(0, 5, txt="Operazione effettuata ai sensi dell'articolo 1, commi da 54 a 89, della Legge n. 190/2014 cos√¨ come modificato dalla Legge n. 208/2015.\nOperazione non soggetta a ritenuta alla fonte a titolo di acconto.\nImposta di bollo assolta sull'originale (se applicabile).")
    
    return pdf.output(dest='S').encode('latin-1')
